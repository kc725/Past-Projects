Module TerminalPrint

module TerminalPrint: sig .. end
val print_map : Objects.Tile.t array array -> int -> unit

print_map map time prints the map map followed by the time time elapsed so far in the world.

val print_living_dead : State.State.t -> unit

print_living_dead state prints information about how many living and dead people there are in the world.

val print_infected : State.State.t -> unit

print_infected state prints the total number of infected people in the world given by state state.

val print_population : State.State.t -> unit

print_population state prints the total number of people in the world given by state state.

val print_world_info : State.State.t -> unit

print_world_info state prints out a bar that indicates the proportion of the population that is living, infected, and dead.
