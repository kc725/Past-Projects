module Controller: sig .. end
type command =
| Quit
| Continue
| Upgrade
| Disease

The possible commands a user can input. If the user's input does not match one of these, an exception is thrown.

exception Empty
exception Malformed
exception Disease_Malformed
val disease_parse : string -> Objects.Disease.t -> Objects.Disease.t

disease_parse str disease parses the string str given through a read_line and either changes one of the disease fields and returns the new disease, exits the disease menu, or raises an error if the input is not formatted correctly.

val parse : string -> command

parse str takes the string str given from a read_line and returns what kind of command the user wishes to run.

val print_disease_menu_numbers : Objects.Disease.t -> Objects.Disease.t

print_disease_menu_number disease prints out information about the given disease disease.

val print_disease_menu : Objects.Disease.t -> Objects.Disease.t

print_disease_menu disease prints out information about the disease disease and prompts the user to make any desired modifications to the disease stats.

val print_upgrade_menu : State.State.t -> State.State.t

print_upgrade_menu state prints out the disease information and allows the user to select any upgrades that they want to make to the disease.
